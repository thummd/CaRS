\section{Introduction}

\begin{frame}{What is FANTOM?}
\textbf{FANTOM}: \textbf{F}ully \textbf{A}utonomous \textbf{N}on-stationary \textbf{T}emporal causal disc\textbf{O}very for \textbf{M}ulti-regime MTS

\vspace{0.5cm}
\textbf{Key Problem:} Causal discovery in multi-regime time series where:
\begin{itemize}
    \item The causal structure changes across different regimes
    \item Regime boundaries are \textbf{unknown} (must be learned)
    \item Data may have \textbf{heteroscedastic} or \textbf{non-Gaussian} noise
\end{itemize}

\vspace{0.5cm}
\textbf{FANTOM's Approach:}
\begin{itemize}
    \item Variational inference with conditional normalizing flows
    \item Joint learning of regime assignments and causal graphs
    \item Augmented Lagrangian optimization for DAG constraint
\end{itemize}
\end{frame}

%==============================================================================
\begin{frame}{What We Are Reproducing}
\textbf{Original Paper:} ``FANTOM: Fully Autonomous Non-stationary Temporal causal discOvery for Multi-regime MTS'' (NeurIPS 2025)

\vspace{0.3cm}
\textbf{Reproduction Scope:}
\begin{itemize}
    \item Synthetic experiments with heteroscedastic noise
    \item Graph sizes: $d \in \{10, 20, 40\}$ nodes
    \item Regime configurations: $K \in \{2, 3\}$ regimes
    \item 3 random seeds per configuration (18 total experiments)
\end{itemize}

\vspace{0.3cm}
\textbf{Metrics Evaluated:}
\begin{itemize}
    \item \textbf{F1 Score} -- Edge detection accuracy
    \item \textbf{SHD} -- Structural Hamming Distance (lower is better)
    \item \textbf{Regime Accuracy} -- Correct regime assignment
\end{itemize}
\end{frame}

\section{Root Cause Analysis}

\begin{frame}{40-Node Issue: Root Cause Analysis}
\textbf{Problem:} Instantaneous F1 is 21-28\% vs paper's 82-86\%

\vspace{0.2cm}
\textbf{Key Discovery:}
\begin{itemize}
    \item Paper's original config uses \texttt{rho = 1.0} for all graph sizes
    \item Our implementation sets \texttt{rho = 0.001} for $n \geq 20$ nodes
    \item Low $\rho$ prevents proper DAG constraint enforcement
\end{itemize}

\vspace{0.2cm}
\textbf{Hyperparameter Sweep (Currently Running):}
\begin{itemize}
    \item Testing \texttt{rho = 1.0} with $\lambda_{\text{sparse}} \in \{50, 100, 150\}$
    \item 10 experiments submitted (6 base + 4 ablation)
    \item Expected completion: 8-12 hours
\end{itemize}

\vspace{0.2cm}
\textbf{What's Working:}
\begin{itemize}
    \item \textcolor{reprogreen}{Lagged edges:} 93.8-95.6\% F1 (exceeds paper's 87.9-91.9\%)
    \item \textcolor{reprogreen}{Regime accuracy:} 96.1-99.9\% (matches paper)
    \item \textcolor{warnred}{Instantaneous edges:} Require $\rho$ fix
\end{itemize}
\end{frame}

\begin{frame}{Why FANTOM? Key Advantages}
\textbf{1. Fully Autonomous:}
\begin{itemize}
    \item No need for pre-specified regime boundaries
    \item Jointly learns structure and regime assignments
\end{itemize}

\vspace{0.2cm}
\textbf{2. Handles Complex Noise:}
\begin{itemize}
    \item Works with heteroscedastic noise (variance changes)
    \item Works with non-Gaussian distributions
    \item Baselines (CASTOR, RPCMCI) fail in these settings
\end{itemize}

\vspace{0.2cm}
\textbf{3. State-of-the-Art Performance:}
\begin{itemize}
    \item Outperforms all baselines on instantaneous edge detection
    \item Near-perfect regime detection accuracy ($>$97\%)
    \item Scales to 40+ node graphs
\end{itemize}

\vspace{0.2cm}
\textbf{4. Principled Approach:}
\begin{itemize}
    \item Variational inference with theoretical guarantees
    \item Conditional normalizing flows for flexible distributions
\end{itemize}
